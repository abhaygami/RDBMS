1. Student (rollno, name, area, city, gender, birthdate)

CREATE TABLE STUDENT (
    ROLLNO INTEGER PRIMARY KEY,
    NAME VARCHAR2(25),
    AREA VARCHAR2(25),
    CITY VARCHAR2(25),
    GENDER VARCHAR2(7) CHECK (GENDER IN ('MALE' , 'FEMALE' , 'OTHER')) ,
    BIRTHDATE DATE
);

2. Marks (rollno, rdbms1, wd1, iot,div)

CREATE TABLE MARKS (
    ROLLNO INTEGER,
    RDBMS1 INTEGER CHECK (RDBMS1 BETWEEN 0 AND 100),
    WD1 INTEGER CHECK (WD1 BETWEEN 0 AND 100),
    IOT INTEGER CHECK (IOT BETWEEN 0 AND 100),
    DIV CHAR CHECK (DIV IN('A','B','C','D')),
    FOREIGN KEY(ROLLNO) REFERENCES STUDENT(ROLLNO)
);

3. Activity (actid, actname, acttype)

CREATE TABLE ACTIVITY (
    ACTID INTEGER PRIMARY KEY,
    ACTNAME VARCHAR2(25),
    ACTTYPE VARCHAR2(25)
);

4. Participant (rollno, actid, actdate, place)

CREATE TABLE PARTICIPANT (
    ROLLNO INTEGER,
    ACTID INTEGER,
    ACTDATE DATE,
    PLACE VARCHAR2(25),
    FOREIGN KEY(ROLLNO) REFERENCES STUDENT(ROLLNO),
    FOREIGN KEY(ACTID) REFERENCES ACTIVITY(ACTID)
);

QUERIES [8]
1. List the student detail who is receiving highest marks in rdbms1

=>
SELECT S.ROLLNO , NAME , RDBMS1
FROM STUDENT S,MARKS M
WHERE S.ROLLNO = M.ROLLNO AND RDBMS1 = (SELECT MAX(RDBMS1) FROM MARKS) ;

2. Display activity which is having maximum participants

=>
SELECT A.ACTID, A.ACTNAME, A.ACTTYPE
FROM ACTIVITY A ,PARTICIPANT P 
WHERE A.ACTID = P.ACTID
GROUP BY A.ACTID, A.ACTNAME, A.ACTTYPE
HAVING COUNT(P.ROLLNO) = (
    SELECT MAX(COUNT(ROLLNO)) 
    FROM PARTICIPANT 
    GROUP BY ACTID
);

3. Display student who is living in ‘vesu’ area and secured more than 70 in iot

=>
SELECT S.ROLLNO , S.NAME , S.AREA , M.IOT
FROM STUDENT S , MARKS M 
WHERE S.ROLLNO = M.ROLLNO AND AREA = 'VESU' AND M.IOT > 70;

4. Display student detail who have secured marks in rdbms1 more than average of b division

=>
SELECT S.ROLLNO , S.NAME , M.RDBMS1 
FROM STUDENT S , MARKS M
WHERE S.ROLLNO = M.ROLLNO AND M.RDBMS1 > (
    SELECT AVG(RDBMS1) 
    FROM MARKS
    WHERE DIV = 'B'
);
